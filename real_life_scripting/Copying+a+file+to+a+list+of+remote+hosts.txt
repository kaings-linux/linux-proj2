Copying a file to a list of remote hosts
==============================================


#!/bin/bash

for HOST in ubuntu01 fedora02 centos03 rhel06
do
	scp somefile $HOST:/var/tmp/
done


# ============ another example ==============

#!/bin/bash

HOSTLIST=`cat /home/user/hostlist`

for HOST in $HOSTLIST
do
	scp somefile $HOST:/var/tmp/
done
